父级属性:
	display: flex
	其他属性
	子元素水平排列
	justify-content {
		center  居中
		space-between 两边对齐
		space-around  拉手排列
		flex-start  居左(默认)
		flex-end  居右
	} 
	子元素垂直排列
	align-item {
		center  居中
		flex-end  底部
		flex-start  顶部
	}
	align-content  同上(多行用)
	
	flex-direction {
		row(默认)  
		column
		*-reverse  反向
	}
	flex-grow {
		1  定义子元素放大比例
	}
	align-self {
		flex-start 覆盖父级align-items属性
	}
	
	order {
		默认0  元素排序 从小到大
	}
